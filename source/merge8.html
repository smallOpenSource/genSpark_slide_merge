Redis에 토큰 저장 및 관리

토큰 ID를 키로, 유효성 정보를 값으로 저장, TTL 설정

토큰 검증 프로세스

로그아웃 요청 시 토큰 상태를 "invalid"로 변경

Prepared Statement 구현 예제

def execute_query(query_name, params=None):
    # SQL 파일 읽기
    sql_file = f"app/sql/{query_name.split('.')[0]}_queries.sql"
    query = ""

    with open(sql_file, 'r') as f:
        content = f.read()
        queries = content.split('-- ')
        for q in queries:
            if q.strip().startswith(query_name.split('.')[1]):
                query = q.replace(query_name.split('.')[1], "").strip()
                break

    # DB 연결 및 쿼리 실행
    conn = get_db_connection()
    try:
        cursor = conn.cursor()
        cursor.execute(query, params) # Prepared Statement 실행
        result = cursor.fetchall()
        conn.commit()
        return result
    finally:
        conn.close()

사용 예시

# 사용자 정보 조회
user_id = 123
user_data = execute_query("user_queries.get_user_by_id", (user_id,))

# 프로필 업데이트
profile_data = "{\"bio\": \"Full-stack developer\"}"
current_time = datetime.now()
execute_query("user_queries.update_user_profile",
           (profile_data, current_time, user_id))

외부 API 연동: OpenAI

# external/openai_service.py
import openai
from flask import current_app

class OpenAIService:
  def __init__(self):
    openai.api_key = current_app.config['OPENAI_API_KEY']

  def generate_response(self, prompt, max_tokens=150):
    try:
      response = openai.Completion.create(
        model="gpt-4",
        prompt=prompt,
        max_tokens=max_tokens
      )
      return response.choices[0].text
    except Exception as e:
      current_app.logger.error(f"OpenAI API 오류: {str(e)}")
      return None

@app.route('/api/weather', methods=['GET'])
def get_weather():
    lat = request.args.get('lat')
    lon = request.args.get('lon')
    
    # Redis 캐시 확인
    cache_key = f"weather:{lat}:{lon}"
    cached = redis.get(cache_key)
    
    if cached:
        return json.loads(cached)
        
    # API 호출 및 캐싱
    result = call_weather_api(lat, lon)
    redis.setex(cache_key, 3600, json.dumps(result))
    return result
